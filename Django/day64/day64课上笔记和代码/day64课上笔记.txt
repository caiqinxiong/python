	# 书
	class Book(models.Model):
		id = models.AutoField(primary_key=True)  # 自增的ID主键
		# 创建一个varchar(64)的唯一的不为空的字段
		title = models.CharField(max_length=64, null=False, unique=True)
		# 和出版社关联的外键字段
		publisher = models.ForeignKey(to="Publisher")
